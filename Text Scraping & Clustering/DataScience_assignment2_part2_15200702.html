porter = PorterStemmer() #Using porter stemmer to output tokens
tokenize = CountVectorizer().build_tokenizer()
from sklearn.feature_extraction import text
stopwords = text.ENGLISH_STOP_WORDS
all_tokens=[]

# List of path all the article files
Files = (glob.glob("*.txt"))   
for f in Files:
    fd = open(str(f), "r")
    text = fd.read()
    tokens1 = tokenize(text.lower()) #Split the files into tokens
    newtokens = []
    for token in tokens1:
        if not token in stopwords and token.isalpha():
            #Removing the stop words and numerics
            token=porter.stem(token)
            newtokens.append(token)
    all_tokens.append(newtokens)
    #all_tokens is a list of preprocessed words of each text file as seperate list items
    fd.close()
